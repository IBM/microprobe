def read_profile(arch, filename):
    """

    :param arch:
    :param filename:

    """
    file_fd = open(filename, "r")
    profile = {}
    for line in file_fd:
        instr, weight = line.split()
        assert arch.isa[instr] not in profile, "Key already found"
        profile[arch.isa[instr]] = int(weight)

    return profile
